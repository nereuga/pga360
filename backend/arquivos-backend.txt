from fastapi import HTTPException, Security, Depends
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer
from supabase import create_client, Client
from sqlalchemy.orm import Session
import os
import models, database

# Configurações do Supabase extraídas do .env
SUPABASE_URL = os.getenv("SUPABASE_URL")
SUPABASE_KEY = os.getenv("SUPABASE_KEY")
supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY)

security = HTTPBearer()

async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Security(security),
    db: Session = Depends(database.get_db)
):
    token = credentials.credentials
    try:
        # Busca o usuário no Supabase
        user_res = supabase.auth.get_user(token)
        supabase_user = user_res.user
        
        # VALIDAÇÃO EXTRA: O e-mail foi confirmado?
        if not supabase_user.email_confirmed_at:
            raise HTTPException(
                status_code=403, 
                detail="E-mail ainda não confirmado. Verifique sua caixa de entrada (Etapa 1)."
            )
        
        # VALIDAÇÃO ETAPA 2: Aprovação manual na base local
        perfil = db.query(models.UsuarioPerfil).filter(
            models.UsuarioPerfil.supabase_uid == supabase_user.id
        ).first()

        if not perfil or not perfil.is_aprovado:
            raise HTTPException(
                status_code=403, 
                detail="Aguardando aprovação manual do administrador (Etapa 2)."
            )

        return perfil 
    except Exception as e:
        # Se o erro já for uma HTTPException (como as de cima), repassa ela
        if isinstance(e, HTTPException): raise e
        # Se for erro de token expirado ou inválido
        raise HTTPException(status_code=401, detail="Sessão inválida ou expirada.")

@app.get("/importacoes", response_model=List[schemas.DiDueListSchema])
def list_importacoes(
    data_reg_ini: Optional[date] = None, data_reg_fim: Optional[date] = None,
    numero: Optional[str] = None, regime: Optional[str] = None,
    modal: Optional[str] = None, skip: int = 0, limit: int = 30,
    db: Session = Depends(database.get_db),
    current_user: models.UsuarioPerfil = Depends(get_current_user)
):
    query = db.query(models.DiDue)
    if data_reg_ini: query = query.filter(models.DiDue.data_registro >= data_reg_ini)
    if data_reg_fim: query = query.filter(models.DiDue.data_registro <= data_reg_fim)
    if numero: query = query.filter(models.DiDue.di_due.ilike(f"%{numero}%"))
    if regime: query = query.filter(models.DiDue.regime == regime)
    if modal: query = query.filter(models.DiDue.modal == modal)
    return query.order_by(models.DiDue.data_registro.desc()).offset(skip).limit(limit).all()

@app.post("/ask-ai", response_model=schemas.AskAiResponse)
def ask_ai(payload: schemas.AskAiRequest, db: Session = Depends(database.get_db), current_user: models.UsuarioPerfil = Depends(get_current_user)):
    model = get_ai_model()
    
    # 1. Recupera Memória e Regras
    memoria = db.query(models.AiMemory).order_by(models.AiMemory.usage_count.desc()).limit(5).all()
    sinonimos = db.query(models.Sinonimos).all()
    kpis = db.query(models.KpiDefinitions).all()
    
    prompt_base = f"""
    Você é um especialista em SQL para o sistema COMEX PGA-360.
    Tabelas principais: 'di_due' e 'item_di_due'.
    Histórico de sucesso: {str([{"q": m.question, "sql": m.sql_query} for m in memoria])}
    Regras de Negócio: {str([{"campo": s.campo_tecnico, "termos": s.termos} for s in sinonimos])}
    
    Instrução: Retorne APENAS o SQL PostgreSQL puro. Use ILIKE para textos.
    Pergunta: {payload.question}
    """
    
    try:
        # 2. Gera e executa SQL
        sql_gerado = model.generate_content(prompt_base).text.strip().replace("```sql", "").replace("```", "")
        
        try:
            result = db.execute(text(sql_gerado))
            rows = [dict(r._mapping) for r in result]
        except Exception as sql_err:
            # AUTO-CORREÇÃO (Self-healing)
            fix_prompt = f"O SQL {sql_gerado} falhou com erro {str(sql_err)}. Corrija o SQL."
            sql_gerado = model.generate_content(fix_prompt).text.strip().replace("```sql", "").replace("```", "")
            result = db.execute(text(sql_gerado))
            rows = [dict(r._mapping) for r in result]

        # 3. Atualiza Memória de Sucesso
        if rows:
            mem = db.query(models.AiMemory).filter(models.AiMemory.question == payload.question).first()
            if mem: mem.usage_count += 1
            else: db.add(models.AiMemory(question=payload.question, sql_query=sql_gerado))
            db.commit()

        # 4. Humanização da Resposta
        final_ans = model.generate_content(f"O banco retornou: {str(rows)}. Explique isso estrategicamente para a pergunta: {payload.question}").text
        return {"answer": final_ans, "sql": sql_gerado}

    except Exception as e:
        return {"answer": f"Erro analítico: {str(e)}", "sql": "falhou"}

    engine = database.engine
    connected = False
    for i in range(10):
        try:
            with engine.connect() as conn:
                conn.execute(text("SELECT 1"))
                connected = True
                break
        except:
            logger.info("⏳ Aguardando banco PostgreSQL...")
            time.sleep(3)
    
    if not connected:
        logger.error("❌ Falha ao conectar no PostgreSQL.")
        return

    # 2. Cria estrutura de tabelas
    models.Base.metadata.create_all(bind=engine)
    
    # 3. Executa as cargas
    db = database.SessionLocal()

        sync_auxiliary_data(db)

        logger.info("🏁 Processo de carga finalizado com sucesso!")
    finally:
        db.close()

import io
import pandas as pd
from datetime import datetime
from reportlab.lib.pagesizes import A4, landscape
from reportlab.platypus import SimpleDocTemplate, Table, TableStyle, Paragraph, Spacer
from reportlab.lib import colors
from reportlab.lib.styles import getSampleStyleSheet

class ReportEngine:
    @staticmethod
    def generate_excel(data: list) -> io.BytesIO:
        """Converte lista de dicionários em arquivo Excel formatado"""
        df = pd.DataFrame(data)
        output = io.BytesIO()
        # openpyxl é a biblioteca que permite criar arquivos .xlsx
        with pd.ExcelWriter(output, engine='openpyxl') as writer:
            df.to_excel(writer, index=False, sheet_name='PGA-360-Export')
        output.seek(0)
        return output

    @staticmethod
    def generate_pdf(data: list, title: str) -> io.BytesIO:
        """Gera um PDF corporativo com tabelas formatadas"""
        output = io.BytesIO()
        # Usamos landscape para caber mais colunas
        doc = SimpleDocTemplate(output, pagesize=landscape(A4), 
                               rightMargin=30, leftMargin=30, topMargin=30, bottomMargin=18)
        elements = []
        styles = getSampleStyleSheet()

        # Cabeçalho do Relatório
        elements.append(Paragraph(f"<b>SISTEMA PGA-360 - GESTÃO ADUANEIRA</b>", styles['Heading1']))
        elements.append(Paragraph(f"Relatório: {title}", styles['Heading2']))
        elements.append(Paragraph(f"Gerado em: {datetime.now().strftime('%d/%m/%Y %H:%M')}", styles['Normal']))
        elements.append(Spacer(1, 20))

        if not data:
            elements.append(Paragraph("Nenhum dado encontrado para exportação.", styles['Normal']))
        else:
            # Extração de colunas e dados
            headers = list(data[0].keys())
            # Formata os nomes das colunas (remove underscores e capitaliza)
            display_headers = [h.replace('_', ' ').upper() for h in headers]
            
            table_data = [display_headers]
            for item in data:
                table_data.append([str(item.get(h, "")) for h in headers])

            # Criação da Tabela com ajuste automático de largura
            # Nota: Em relatórios muito largos, pode ser necessário limitar as colunas
            t = Table(table_data, repeatRows=1)
            
            # Estilização da Tabela (Padrão Corporativo Azul Marinho)
            t.setStyle(TableStyle([
                ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor("#1A3A5A")),
                ('TEXTCOLOR', (0, 0), (-1, 0), colors.whitesmoke),
                ('ALIGN', (0, 0), (-1, -1), 'LEFT'),
                ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
                ('FONTSIZE', (0, 0), (-1, -1), 7), # Fonte menor para caber tudo
                ('BOTTOMPADDING', (0, 0), (-1, 0), 10),
                ('BACKGROUND', (0, 1), (-1, -1), colors.whitesmoke),
                ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
                ('VALIGN', (0, 0), (-1, -1), 'MIDDLE'),
            ]))
            elements.append(t)

        doc.build(elements)
        output.seek(0)
        return output
